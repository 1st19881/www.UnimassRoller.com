  109.   /**
    110. 

  110.    * Connect to the server specified on the port specified.
    111. 

  111.    * If the port is not specified use the default SMTP_PORT.
    112. 

  112.    * If tval is specified then a connection will try and be
    113. 

  113.    * established with the server for that number of seconds.
    114. 

  114.    * If tval is not specified the default is 30 seconds to
    115. 

  115.    * try on the connection.
    116. 

  116.    *
    117. 

  117.    * SMTP CODE SUCCESS: 220
    118. 

  118.    * SMTP CODE FAILURE: 421
    119. 

  119.    * @access public
    120. 

  120.    * @return bool 
    121. 

  121.    */

  122.   public function Connect($host, $port = 0, $tval = 30) {
    123. 

  123.     // set the error val to null so there is no confusion
    124. 

  124.     $this->error = null;
    125. 

  125.  
    126. 

  126.     // make sure we are __not__ connected
    127. 

  127.     if($this->connected()) {
    128. 

  128.       // already connected, generate error
    129. 

  129.       $this->error = array("error" => "Already connected to a server");
    130. 

  130.       return false;
    131. 

  131.     }
    132. 

  132.  
    133. 

  133.     if(empty($port)) {
    134. 

  134.       $port = $this->SMTP_PORT;
    135. 

  135.     }
    136. 

  136.  
    137. 

  137.     // connect to the smtp server
    138. 

  138.     $this->smtp_conn = @fsockopen($host,    // the host of the server
    139. 

  139.                                  $port,    // the port to use
    140. 

  140.                                  $errno,   // error number if any
    141. 

  141.                                  $errstr,  // error message if any
    142. 

  142.                                  $tval);   // give up after ? secs
    143. 

  143.     // verify we connected properly
    144. 

  144.     if(empty($this->smtp_conn)) {
    145. 

  145.       $this->error = array("error" => "Failed to connect to server",
    146. 

  146.                            "errno" => $errno,
    147. 

  147.                            "errstr" => $errstr);
    148. 

  148.       if($this->do_debug >= 1) {
    149. 

  149.         echo "SMTP -> ERROR: " . $this->error["error"] . ": $errstr ($errno)" . $this->CRLF . '<br />';
    150. 

  150.       }
    151. 

  151.       return false;
    152. 

  152.     }
    153. 

  153.  
    154. 

  154.     // SMTP server can take longer to respond, give longer timeout for first read
    155. 

  155.     // Windows does not have support for this timeout function
    156. 

  156.     if(substr(PHP_OS, 0, 3) != "WIN")
    157. 

  157.      socket_set_timeout($this->smtp_conn, $tval, 0);
    158. 

  158.  
    159. 

  159.     // get any announcement
    160. 

  160.     $announce = $this->get_lines();
    161. 

  161.  
    162. 

  162.     if($this->do_debug >= 2) {
    163. 

  163.       echo "SMTP -> FROM SERVER:" . $announce . $this->CRLF . '<br />';
    164. 

  164.     }
    165. 

  165.  
    166. 

  166.     return true;
    167. 

  167.   }

  178.   public function StartTLS() {
    179. 

  179.     $this->error = null; # to avoid confusion
    180. 

  180.  
    181. 

  181.     if(!$this->connected()) {
    182. 

  182.       $this->error = array("error" => "Called StartTLS() without being connected");
    183. 

  183.       return false;
    184. 

  184.     }
    185. 

  185.  
    186. 

  186.     fputs($this->smtp_conn,"STARTTLS" . $this->CRLF);
    187. 

  187.  
    188. 

  188.     $rply = $this->get_lines();
    189. 

  189.     $code = substr($rply,0,3);
    190. 

  190.  
    191. 

  191.     if($this->do_debug >= 2) {
    192. 

  192.       echo "SMTP -> FROM SERVER:" . $rply . $this->CRLF . '<br />';
    193. 

  193.     }
    194. 

  194.  
    195. 

  195.     if($code != 220) {
    196. 

  196.       $this->error =
    197. 

  197.          array("error"     => "STARTTLS not accepted from server",
    198. 

  198.                "smtp_code" => $code,
    199. 

  199.                "smtp_msg"  => substr($rply,4));
    200. 

  200.       if($this->do_debug >= 1) {
    201. 

  201.         echo "SMTP -> ERROR: " . $this->error["error"] . ": " . $rply . $this->CRLF . '<br />';
    202. 

  202.       }
    203. 

  203.       return false;
    204. 

  204.     }
    205. 

  205.  
    206. 

  206.     // Begin encrypted connection
    207. 

  207.     if(!stream_socket_enable_crypto($this->smtp_conn, true, STREAM_CRYPTO_METHOD_TLS_CLIENT)) {
    208. 

  208.       return false;
    209. 

  209.     }
    210. 

  210.  
    211. 

  211.     return true;
    212. 

  212.   }

  227.     if($code != 334) {

    245. 

  245.       $this->error =
    246. 

  246.         array("error" => "Username not accepted from server",
    247. 

  247.               "smtp_code" => $code,
    248. 

  248.               "smtp_msg" => substr($rply,4));

  249.       if($this->do_debug >= 1) {

  254.

  257.

  258.     $rply = $this->get_lines();
    259. 

  259.     $code = substr($rply,0,3);

  260.  
    261. 

  261.     if($code != 235) {
    262. 

  262.       $this->error =
    263. 

  263.         array("error" => "Password not accepted from server",
    264. 

  264.               "smtp_code" => $code,
    265. 

  265.               "smtp_msg" => substr($rply,4));
    266. 

  266.       if($this->do_debug >= 1) {
    267. 

  267.         echo "SMTP -> ERROR: " . $this->error["error"] . ": " . $rply . $this->CRLF . '<br />';
    268. 

  268.       }
    269. 

  269.       return false;
    270. 

  270.     }
    271. 

  271.  
    272. 

  272.     return true;
    273. 

  273.   }

  280.   public function Connected() {
    281. 

  281.     if(!empty($this->smtp_conn)) {
    282. 

  282.       $sock_status = socket_get_status($this->smtp_conn);
    283. 

  283.       if($sock_status["eof"]) {
    284. 

  284.         // the socket is valid but we are not connected
    285. 

  285.         if($this->do_debug >= 1) {
    286. 

  286.             echo "SMTP -> NOTICE:" . $this->CRLF . "EOF caught while checking if connected";
    287. 

  287.         }
    288. 

  288.         $this->Close();
    289. 

  289.         return false;
    290. 

  290.       }
    291. 

  291.       return true; // everything looks good
    292. 

  292.     }
    293. 

  293.     return false;
    294. 

  294.   }

  319.   /**
    320. 

  320.    * Issues a data command and sends the msg_data to the server
    321. 

  321.    * finializing the mail transaction. $msg_data is the message
    322. 

  322.    * that is to be send with the headers. Each header needs to be
    323. 

  323.    * on a single line followed by a <CRLF> with the message headers
    324. 

  324.    * and the message body being seperated by and additional <CRLF>.
    325. 

  325.    *
    326. 

  326.    * Implements rfc 821: DATA <CRLF>
    327. 

  327.    *
    328. 

  328.    * SMTP CODE INTERMEDIATE: 354
    329. 

  329.    *     [data]
    330. 

  330.    *     <CRLF>.<CRLF>
    331. 

  331.    *     SMTP CODE SUCCESS: 250
    332. 

  332.    *     SMTP CODE FAILURE: 552,554,451,452
    333. 

  333.    * SMTP CODE FAILURE: 451,554
    334. 

  334.    * SMTP CODE ERROR  : 500,501,503,421
    335. 

  335.    * @access public
    336. 

  336.    * @return bool 
    337. 

  337.    */
    338. 

  338.   public function Data($msg_data) {
    339. 

  339.     $this->error = null; // so no confusion is caused
    340. 

  340.  
    341. 

  341.     if(!$this->connected()) {
    342. 

  342.       $this->error = array(
    343. 

  343.               "error" => "Called Data() without being connected");
    344. 

  344.       return false;
    345. 

  345.     }
    346. 

  346.  
    347. 

  347.     fputs($this->smtp_conn,"DATA" . $this->CRLF);
    348. 

  348.  
    349. 

  349.     $rply = $this->get_lines();
    350. 

  350.     $code = substr($rply,0,3);
    351. 

  351.  
    352. 

  352.     if($this->do_debug >= 2) {
    353. 

  353.       echo "SMTP -> FROM SERVER:" . $rply . $this->CRLF . '<br />';
    354. 

  354.     }
    355. 

  355.  
    356. 

  356.     if($code != 354) {
    357. 

  357.       $this->error =
    358. 

  358.         array("error" => "DATA command not accepted from server",
    359. 

  359.               "smtp_code" => $code,
    360. 

  360.               "smtp_msg" => substr($rply,4));
    361. 

  361.       if($this->do_debug >= 1) {
    362. 

  362.         echo "SMTP -> ERROR: " . $this->error["error"] . ": " . $rply . $this->CRLF . '<br />';
    363. 

  363.       }
    364. 

  364.       return false;
    365. 

  365.     }
    366. 

  366.  
    367. 

  367.     /* the server is ready to accept data!
    368. 

  368.      * according to rfc 821 we should not send more than 1000
    369. 

  369.      * including the CRLF
    370. 

  370.      * characters on a single line so we will break the data up
    371. 

  371.      * into lines by \r and/or \n then if needed we will break
    372. 

  372.      * each of those into smaller lines to fit within the limit.
    373. 

  373.      * in addition we will be looking for lines that start with
    374. 

  374.      * a period '.' and append and additional period '.' to that
    375. 

  375.      * line. NOTE: this does not count towards limit.
    376. 

  376.      */
    377. 

  377.  
    378. 

  378.     // normalize the line breaks so we know the explode works
    379. 

  379.     $msg_data = str_replace("\r\n","\n",$msg_data);
    380. 

  380.     $msg_data = str_replace("\r","\n",$msg_data);
    381. 

  381.     $lines = explode("\n",$msg_data);
    382. 

  382.  
    383. 

  383.     /* we need to find a good way to determine is headers are
    384. 

  384.      * in the msg_data or if it is a straight msg body
    385. 

  385.      * currently I am assuming rfc 822 definitions of msg headers
    386. 

  386.      * and if the first field of the first line (':' sperated)
    387. 

  387.      * does not contain a space then it _should_ be a header
    388. 

  388.      * and we can process all lines before a blank "" line as
    389. 

  389.      * headers.
    390. 

  390.      */
    391. 

  391.  
    392. 

  392.     $field = substr($lines[0],0,strpos($lines[0],":"));
    393. 

  393.     $in_headers = false;
    394. 

  394.     if(!empty($field) && !strstr($field," ")) {
    395. 

  395.       $in_headers = true;
    396. 

  396.     }
    397. 

  397.  
    398. 

  398.     $max_line_length = 998; // used below; set here for ease in change
    399. 

  399.  
    400. 

  400.     while(list(,$line) = @each($lines)) {
    401. 

  401.       $lines_out = null;
    402. 

  402.       if($line == "" && $in_headers) {
    403. 

  403.         $in_headers = false;
    404. 

  404.       }
    405. 

  405.       // ok we need to break this line up into several smaller lines
    406. 

  406.       while(strlen($line) > $max_line_length) {
    407. 

  407.         $pos = strrpos(substr($line,0,$max_line_length)," ");
    408. 

  408.  
    409. 

  409.         // Patch to fix DOS attack
    410. 

  410.         if(!$pos) {
    411. 

  411.           $pos = $max_line_length - 1;
    412. 

  412.           $lines_out[] = substr($line,0,$pos);
    413. 

  413.           $line = substr($line,$pos);
    414. 

  414.         } else {
    415. 

  415.           $lines_out[] = substr($line,0,$pos);
    416. 

  416.           $line = substr($line,$pos + 1);
    417. 

  417.         }
    418. 

  418.  
    419. 

  419.         /* if processing headers add a LWSP-char to the front of new line
    420. 

  420.          * rfc 822 on long msg headers
    421. 

  421.          */
    422. 

  422.         if($in_headers) {
    423. 

  423.           $line = "\t" . $line;
    424. 

  424.         }
    425. 

  425.       }
    426. 

  426.       $lines_out[] = $line;
    427. 

  427.  
    428. 

  428.       // send the lines to the server
    429. 

  429.       while(list(,$line_out) = @each($lines_out)) {
    430. 

  430.         if(strlen($line_out) > 0)
    431. 

  431.         {
    432. 

  432.           if(substr($line_out, 0, 1) == ".") {
    433. 

  433.             $line_out = "." . $line_out;
    434. 

  434.           }
    435. 

  435.         }
    436.
